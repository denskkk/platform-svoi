# ⚡ Швидке виправлення - копіюйте ці команди

# 1. Stash локальних змін та оновлення коду
cd /var/www/sviydlyasvoih/platform-svoi
git stash
git pull origin main

# 2. Видалення старого білду та ребілд
rm -rf .next
npx prisma generate
npm run build

# 3. Перезапуск sviy-platform (не sviy-web!)
pm2 delete sviy-web
pm2 restart sviy-platform

# 4. Перевірка статусу
pm2 status
pm2 logs sviy-platform --lines 30 --nostream
